## Q3 (a)

```asm
INCLUDE Irvine32.inc

.code
    arr DWORD 10,25,100,35,30,85,90,84,72,60
main PROC
    mov esi,OFFSET arr
    mov eax,0
    call MAXIMUM
    call WriteDec
    exit
main ENDP

MAXIMUM PROC
    mov ecx,LENGTHOF arr
    mov ebx,0
L1:
    cmp [esi+ebx],eax
    JLE next
    mov eax,[esi+ebx]
    add ebx,4
    loop L1
next:
    add ebx,4
    loop L1

    ret
MAXIMUM ENDP
END main
```
