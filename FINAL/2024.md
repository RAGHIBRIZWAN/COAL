## Q4

```asm
TITLE Convert Binary to Decimal
INCLUDE Irvine32.inc

.data
    arr BYTE 0,0,1,0,1,0,0,0   ; Binary = 00101000 = 40

.code
main PROC
    mov esi, 0                ; index for array
    mov ecx, LENGTHOF arr     ; number of bits (8)
    mov eax, 0                ; will hold final decimal value

L1:
    shl eax, 1                ; shift left to make room for next bit
    mov bl, arr[esi]          ; get current bit (0 or 1)
    add al, bl                ; add bit to eax
    inc esi                   ; move to next bit
    loop L1

    call WriteDec             ; display the final decimal value
    call Crlf
    exit
main ENDP
END main
```

## Q6

```asm
TITLE WORD COUNT
INCLUDE Irvine32.inc

.data
    arr BYTE "Nothing is worth more than this day",0  

.code
main PROC
    mov ecx,LENGTHOF arr
    mov esi,0
    mov ebx,1
    mov dl,0
L1:
    mov al,arr[esi]
    cmp al,0
    je quit
    cmp arr[esi+1],dl
    je quit
    cmp al,20h
    jne next
    inc ebx
    inc esi
    loop L1
    jmp quit

next:
    inc esi
    loop L1

quit:
    mov eax,ebx
    call WriteDec
    exit
main ENDP
END main
```
