```asm
INCLUDE Irvine32.inc

.data
	Str1 BYTE '127&j~3#^&*#*#45^',0
	print1 BYTE "FOUND",0
	print2 BYTE "NOT FOUND",0
	count DWORD 0
.code
main PROC
	mov edi, OFFSET Str1
	mov al, '#' 
	call Scan_String
	
exit
main ENDP

Scan_String PROC
	cld
	mov ecx,LENGTHOF Str1
	mov ebx,count
L1:
	SCASB
	je return
	inc ebx
	loop L1
	
	mov edx, OFFSET print2
    call WriteString
    call Crlf
    ret

return:
	mov edx,OFFSET print1
	call WriteString
	call Crlf
	mov eax,ebx
	call WriteDec
	ret
Scan_String ENDP
END main
```
