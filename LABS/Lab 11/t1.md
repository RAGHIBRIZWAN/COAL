```asm
INCLUDE Irvine32.inc

.code
main PROC
	mov eax,5
	mov ebx, eax    ; EBX = EAX (original value)
	shl eax, 4      ; EAX = EAX * 16 (2^4)
	add eax, ebx    ; EAX = (EAX * 16) + EAX = EAX * 17
	shl ebx, 2      ; EBX = EAX * 4 (2^2)
	add eax, ebx    ; EAX = (EAX * 17) + (EAX * 4) = EAX * 21
main ENDP
END main
```
