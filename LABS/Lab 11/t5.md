```asm
INCLUDE Irvine32.inc

.data
num1 QWORD 0FFFFFFFF12345678h   
num2 QWORD 0000000011111111h    
result QWORD ?                  

.code
main PROC
    mov  esi, OFFSET num1       
    mov  edi, OFFSET num2       
    mov  ebx, OFFSET result     
    call Extended_Add           

    mov  eax, DWORD PTR [ebx+4] 
    call WriteHex               
    mov  eax, DWORD PTR [ebx]   
    call WriteHex               
    call Crlf

    exit
main ENDP

Extended_Add PROC

    mov  eax, [esi]      
    add  eax, [edi]      
    mov  [ebx], eax      

    mov  eax, [esi+4]    
    adc  eax, [edi+4]    
    mov  [ebx+4], eax    

    ret
Extended_Add ENDP

END main
```
