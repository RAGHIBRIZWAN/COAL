```asm
INCLUDE Irvine32.inc

.data
	v1 DWORD ?
	v2 DWORD ?
	v3 DWORD ?
	v4 DWORD ?
	msg BYTE "Enter the number: ",0
	lose BYTE "They are not equal",0
	win BYTE "They are equal",0
.code
main PROC
	mov edx,OFFSET msg
	call WriteString
	call ReadInt
	mov v1,eax

	mov edx,OFFSET msg
	call WriteString
	call ReadInt
	mov v2,eax

	mov edx,OFFSET msg
	call WriteString
	call ReadInt
	mov v3,eax

	mov edx,OFFSET msg
	call WriteString
	call ReadInt
	mov v4,eax

	mov eax,v1
	cmp eax,v2
	JNE next

	mov eax,v2
	cmp eax,v3
	JNE next

	mov eax,v3
	cmp eax,v4
	JNE next

	mov edx, OFFSET win
	call WriteString
	exit
next:
	mov edx, OFFSET lose
	call WriteString
	exit
main ENDP
END main
```
