```asm
INCLUDE Irvine32.inc

.data
	intArr SWORD 0, 0, 0, 150, 120, 35, -12, 66, 4, 0
	msg BYTE "Element found: ",0
.code
main PROC
	mov esi,0
	mov ecx, LENGTHOF intArr
L1:
	mov ax,intArr[esi]
	cmp ax,0
	JNE next
	add esi,2
	loop L1

next:
	mov edx,OFFSET msg
	call WriteString
	movzx eax, intArr[esi]
	call WriteDec
exit
main ENDP
END main
```
