```asm
INCLUDE Irvine32.inc

.data
	var DWORD 5
	x DWORD ?
.code
main PROC
	mov ecx, 10
	mov edx,var+1
	mov ebx,var

	cmp ebx,ecx
	JL checkAnd

	mov eax,1
	mov x,eax
	jmp ending

checkAnd:
	cmp ecx, edx
	JGE final

	mov eax,1
	mov x, eax
	jmp ending

final:
	mov eax,0
	mov x,eax
	jmp ending

ending:
	call WriteDec
	exit
main ENDP
END main
```
