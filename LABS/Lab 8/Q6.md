```asm
INCLUDE Irvine32.inc   

.data
    array WORD 10, 4, 7, 14, 299, 156, 3, 19, 29, 300, 20   
    arrSize DWORD 11                    

.code
main PROC
    mov ecx, arrSize                
    dec ecx                    

L1:
    mov esi, 0                 
    
L2:
    mov ax, [array + esi*2]   
    mov dx, [array + esi*2 + 2] 
    cmp ax, dx                
    jle noSwap                 
    
    mov [array + esi*2], dx    
    mov [array + esi*2 + 2], ax 
    
noSwap:
    inc esi
    cmp esi, ecx                
    jl L2

    dec ecx                     
    jnz L1

    mov esi, 0                   

print:
    movzx eax, [array + esi*2]     
    call WriteInt                
    call Crlf                    
    inc esi                       
    cmp esi, arrSize                    
    jl print

    exit                          
main ENDP 
END main
```
