```asm
INCLUDE Irvine32.inc

.data
	a1 DWORD 2
	a2 DWORD 3
	a3 DWORD 4
.code
main PROC
	push a3
	push a2 
	push a1
	call ThreeProd
exit
main ENDP

ThreeProd PROC
	push ebp
	mov ebp,esp
	mov eax,[ebp+8]
	imul eax,[ebp+12]
	imul eax,[ebp+16]
	call WriteDec
	pop ebp
	ret 12
ThreeProd ENDP
END main
```
