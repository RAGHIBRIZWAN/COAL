```asm
INCLUDE Irvine32.inc   

.data
    array DWORD 10, 4, 7, 14, 0, 299, 156, 3, 19, 29, 300, 20   
    arrSize DWORD 12                

.code
BubbleSort PROTO, arr:PTR DWORD
main PROC
    INVOKE BubbleSort,ADDR array
exit                       
main ENDP 


BubbleSort PROC, arr:PTR DWORD
    mov ecx,arrSize
    dec ecx

L1:
    push ecx
    mov esi,arr
L2:
    mov eax,[esi]
    mov ebx,[esi+4]
    cmp eax,ebx
    jle next

    mov [esi],ebx
    mov [esi+4],eax
    
next:
    add esi,4
    loop L2

    pop ecx
    loop L1

    mov ecx,arrSize
    mov esi,arr
print:
    mov eax,[esi]
    call WriteDec
    call Crlf
    add esi,4
    loop Print
    ret
BubbleSort ENDP
END main
```
